# Generic

- 타입을 파라미터처럼 받아서 활용

```
//유니온으로 매개변수 전달은 가독성 떨어짐 -> 제네릭 사용
function getSize<T>(arr: T[]):number{
    return arr.length;
}

const arr1 = [1,2,3];
getSize<number>(arr1); // 제네릭 사용

const arr2 = ["a","b","c"];
getSize(arr2);
```

```
//하나의 인터페이스로 여러 모습의 객체 만들기
interface Mobile<T>{
    name: string;
    price: number;
    option: T;
}

const m1: Mobile<object>{ //{colr:string; coupon:boolean;}
    name: "s21",
    price: 100,
    option: {
        color:"red",
        coupon: false,
    },
};

const m2: Mobile<string>{
    name: "s20",
    price: 100,
    option: "free",
};
```
